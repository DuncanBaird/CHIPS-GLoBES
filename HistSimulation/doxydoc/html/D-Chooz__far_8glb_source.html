45 In example5.c D-Chooz_far.glb and D-Chooz_near.glb are used and the chi-squared
46 function chiDCNorm is defined in the following way:
47 
48  ***************************************************************************
49  * Calculate chi^2 for Double Chooz, including the following systematical *
50  * errors: *
51  * x[0]: Flux normalization of reactor *
52  * x[1]: Fiducial mass error - far detector *
53  * x[2]: Fiducial mass error - near detector *
54  * x[3]: Energy calibration error - far detector *
55  * x[4]: Energy calibration error - near detector *
56  ***************************************************************************
57 
58 double chiDCNorm(int exp, int rule, int n_params, double *x, double *errors,
59  void *user_data)
60 {
61  int n_bins = glbGetNumberOfBins(EXP_FAR);
62  double *true_rates_N = glbGetRuleRatePtr(EXP_NEAR, 0);
63  double *true_rates_F = glbGetRuleRatePtr(EXP_FAR, 0);
64  double signal_fit_rates_N[n_bins];
65  double signal_fit_rates_F[n_bins];
66  double signal_norm_N, signal_norm_F;
67  int ew_low, ew_high;
68  double emin, emax;
69  double fit_rate;
70  double chi2 = 0.0;
71  int i;
72 
73  glbGetEminEmax(exp, &emin, &emax);
74  glbGetEnergyWindowBins(exp, rule, &ew_low, &ew_high);
75 
76  glbShiftEnergyScale(x[3], glbGetSignalFitRatePtr(EXP_FAR, 0),
77  signal_fit_rates_F, n_bins, emin, emax);
78  glbShiftEnergyScale(x[4], glbGetSignalFitRatePtr(EXP_NEAR, 0),
79  signal_fit_rates_N, n_bins, emin, emax);
80 
81  signal_norm_F = 1.0 + x[0] + x[1];
82  signal_norm_N = 1.0 + x[0] + x[2];
83  for (i=ew_low; i <= ew_high; i++)
84  {
85  fit_rate = signal_norm_F * signal_fit_rates_F[i];
86  chi2 += likelihood(true_rates_F[i], fit_rate, true_rates_F[i]);
87 
88  fit_rate = signal_norm_N * signal_fit_rates_N[i];
89  chi2 += likelihood(true_rates_N[i], fit_rate, true_rates_N[i]);
90  }
91 
92  for (i=0; i < n_params; i++)
93  chi2 += square(x[i] / errors[i]);
94 
95  for (i=0; i < n_params; i++)
96  sys_startval[i] = x[i];
97 
98  return chi2;
99 }
