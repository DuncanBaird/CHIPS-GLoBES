89 /* Gauss likelihood (this is sufficient here due to the large event numbers
90  * in a reactor experiment; for other setups, one should use Poisson statistics) */
91 inline double likelihood(double true_rate, double fit_rate, double sqr_sigma)
92 {
93  if (sqr_sigma > 0)
94  return square(true_rate - fit_rate) / sqr_sigma;
95  else
96  return 0.0;

105 /***************************************************************************
106  * Calculate chi^2 for Double Chooz, including the following systematical *
107  * errors: *
108  * x[0]: Flux normalization of reactor *
109  * x[1]: Fiducial mass error - far detector *
110  * x[2]: Fiducial mass error - near detector *
111  * x[3]: Energy calibration error - far detector *
112  * x[4]: Energy calibration error - near detector *
113  ***************************************************************************/
114 double chiDCNorm(int exp, int rule, int n_params, double *x, double *errors,
115  void *user_data){
116  int n_bins = glbGetNumberOfBins(EXP_FAR);
117  double *true_rates_N = glbGetRuleRatePtr(EXP_NEAR, 0);
118  double *true_rates_F = glbGetRuleRatePtr(EXP_FAR, 0);
119  double signal_fit_rates_N[n_bins];
120  double signal_fit_rates_F[n_bins];
121  double signal_norm_N, signal_norm_F;
122  int ew_low, ew_high;
123  double emin, emax;
124  double fit_rate;
125  double chi2 = 0.0;
126  int i;
127 
128  /* Request simulated energy interval and analysis energy window */
129  glbGetEminEmax(exp, &emin, &emax);
130  glbGetEnergyWindowBins(exp, rule, &ew_low, &ew_high);
131 
132  /* Apply energy calibration error */
133  glbShiftEnergyScale(x[3], glbGetSignalFitRatePtr(EXP_FAR, 0),
134  signal_fit_rates_F, n_bins, emin, emax);
135  glbShiftEnergyScale(x[4], glbGetSignalFitRatePtr(EXP_NEAR, 0),
136  signal_fit_rates_N, n_bins, emin, emax);
137 
138  /* Loop over all bins in energy window */
139  signal_norm_F = 1.0 + x[0] + x[1];
140  signal_norm_N = 1.0 + x[0] + x[2];
141  for (i=ew_low; i <= ew_high; i++)
142  {
143  /* Statistical part of chi^2 for far detector
144  * Normalization is affected by flux error x[0] and fiducial mass error x[1] */
145  fit_rate = signal_norm_F * signal_fit_rates_F[i];
146  chi2 += likelihood(true_rates_F[i], fit_rate, true_rates_F[i]);
147 
148  /* Statistical part of chi^2 for near detector
149  * Normalization is affected by flux error x[0] and fiducial mass error x[2] */
150  fit_rate = signal_norm_N * signal_fit_rates_N[i];
151  chi2 += likelihood(true_rates_N[i], fit_rate, true_rates_N[i]);
152  }
153 
154  /* Systematical part of chi^2 (= priors) */
155  for (i=0; i < n_params; i++)
156  chi2 += square(x[i] / errors[i]);
157 
158  /* Save the systematics parameters as starting values for the next step */
159  for (i=0; i < n_params; i++)
160  sys_startval[i] = x[i];
161 
162  return chi2;
163 }

166 /***************************************************************************
167  * Calculate chi^2 for Double Chooz, including the following systematical *
168  * errors: *
169  * x[0]: Flux normalization of reactor *
170  * x[1]: Fiducial mass error - far detector *
171  * x[2]: Fiducial mass error - near detector *
172  * x[3]: Energy calibration error - far detector *
173  * x[4]: Energy calibration error - near detector *
174  * x[5]...x[5+nBins-1]: Spectral error *
175  * and the bin-to-bin error sigma_binbin. The calculation is based on *
176  * Eq. (9) of hep-ph/0303232. *
177  ***************************************************************************/
178 double chiDCSpectral(int exp, int rule, int n_params, double *x, double *errors,
179  void *user_data){
180  int n_bins = glbGetNumberOfBins(EXP_FAR);
181  double *true_rates_N = glbGetRuleRatePtr(EXP_NEAR, 0);
182  double *true_rates_F = glbGetRuleRatePtr(EXP_FAR, 0);
183  double signal_fit_rates_N[n_bins];
184  double signal_fit_rates_F[n_bins];
185  double signal_norm_N, signal_norm_F;
186  int ew_low, ew_high;
187  double emin, emax;
188  double fit_rate;
189  double chi2 = 0.0;
190  int i;
191 
192  /* Request simulated energy interval and analysis energy window */
193  glbGetEminEmax(exp, &emin, &emax);
194  glbGetEnergyWindowBins(exp, rule, &ew_low, &ew_high);
195 
196  /* Apply energy calibration error */
197  glbShiftEnergyScale(x[3], glbGetSignalFitRatePtr(EXP_FAR, 0),
198  signal_fit_rates_F, n_bins, emin, emax);
199  glbShiftEnergyScale(x[4], glbGetSignalFitRatePtr(EXP_NEAR, 0),
200  signal_fit_rates_N, n_bins, emin, emax);
201 
202  /* Loop over all bins in energy window */
203  signal_norm_F = 1.0 + x[0] + x[1];
204  signal_norm_N = 1.0 + x[0] + x[2];
205  for (i=ew_low; i <= ew_high; i++)
206  {
207  /* Statistical part of chi^2 for far detector
208  * Normalization is affected by flux error x[0], fiducial mass error x[1],
209  * spectral perturbations x[5+i], and bin-to-bin error sigma_binbin */
210  fit_rate = (signal_norm_F + x[5+i]) * signal_fit_rates_F[i];
211  chi2 += likelihood( true_rates_F[i], fit_rate,
212  true_rates_F[i] * (1.0 + true_rates_F[i]*square(sigma_binbin)) );
213 
214  /* Statistical part of chi^2 for near detector
215  * Normalization is affected by flux error x[0], fiducial mass error x[2],
216  * spectral perturbations x[5+i], and bin-to-bin error sigma_binbin */
217  fit_rate = (signal_norm_N + x[5+i]) * signal_fit_rates_N[i];
218  chi2 += likelihood( true_rates_N[i], fit_rate,
219  true_rates_N[i] * (1.0 + true_rates_N[i]*square(sigma_binbin)) );
220  }
221 
222  /* Systematical part of chi^2 (= priors) */
223  for (i=0; i < n_params; i++)
224  chi2 += square(x[i] / errors[i]);
225 
226  /* Save the systematics parameters as starting values for the next step */
227  for (i=0; i < n_params; i++)
228  sys_startval[i] = x[i];
229 
230  return chi2;
231 }

238 /* Callback function for GSL root finder */
239 double DoChiSquare(double x, void *dummy){
240  double thetheta13, chi2;
241 
242  /* Set vector of test values */
243  thetheta13 = asin(sqrt(pow(10,x)))/2;
244  glbSetOscParams(test_values, thetheta13, GLB_THETA_13);
245  glbSetOscParams(test_values, 200.0/2 * (x+4)*M_PI/180, GLB_DELTA_CP);
246 
247  /* Set starting values for systematics minimiyer to the coordinates of
248  * minimum in the last iteration. This accelerates the minimization and
249  * prevents convergence problems. */
250  glbSetSysStartingValuesList(EXP_FAR, 0, GLB_ON, sys_startval);
251 
252  /* Compute Chi^2 for all loaded experiments and all rules
253  * Correlations are unimportant in reactor experiments, so glbChiSys is sufficient */
254  chi2 = glbChiSys(test_values, GLB_ALL, GLB_ALL);
255 
256  return chi2 - chi2_goal;
257 }

273 /* Compute sensitivity to sin^2(2*th13) for different running times */
274 void ComputeSensitivityCurve(){
275  double t_factor = pow(max_runtime/min_runtime, 1.0/tSteps);
276  double t;
277  double x; /* Current value of log[sin^2(2*th13)] */
278  double x_lo, x_hi; /* Bracketing interval for root finder */
279  double x_sens[tSteps+1];/* Calculated sensitivities to log[sin^2(2*th13)] */
280  int gsl_status; /* GSL error code */
281  int iter; /* Iteration counter for root finder */
282  const int max_iter=100; /* Maximum number of iterations allowed in root finder */
283  int j;
284 
285  for (j=0; j <= tSteps; j++) /* Initialize output vector */
286  x_sens[j] = 1; /* As x is always <= 0 this signals "no value" */
287 
288  /* Loop over all data points in this dataset, using log stepping */
289  for (j=0; j <= tSteps; j++)
290  {
291  /* Set running time */
292  t = min_runtime * pow(t_factor,j);
293  glbSetRunningTime(EXP_FAR, 0, t);
294  glbSetRunningTime(EXP_NEAR, 0, t);
295 
296  /* Calculate "true" event rates */
297  glbDefineParams(true_values,theta12,theta13,theta23,deltacp,sdm,ldm);
298  glbSetDensityParams(true_values,1.0,GLB_ALL);
299  glbDefineParams(test_values,theta12,theta13,theta23,deltacp,sdm,ldm);
300  glbSetDensityParams(test_values,1.0,GLB_ALL);
301  glbDefineParams(input_errors, 0.1*theta12, 0, 0.15*theta23, 0, 0.05*sdm, 0.05*ldm);
302  glbSetDensityParams(input_errors, 0.05, GLB_ALL);
303  glbSetOscillationParameters(true_values);
304  glbSetInputErrors(input_errors);
305  glbSetRates();
306 
307  /* Determine sensitivity to sin^2(2*th13) by using the GSL Brent-Dekker
308  * algorithm to find the th13, for which chi^2 = 2.7 (90%) */
309  x_lo = -3.0;
310  x_hi = -0.1;
311  iter = 0;
312 
313  /* Start root finder. The initial search interval is guessed based on the
314  * results from a neighboring grid point */
315  double deviation=0.1;
316  do {
317  if (j > 0) {
318  x_lo = x_sens[j-1] - deviation;
319  x_hi = min(x_sens[j-1] + deviation, -0.001);
320  }
321  gsl_status = gsl_root_fsolver_set(s, &gsl_func, x_lo, x_hi);
322  deviation *= 2;
323  } while (gsl_status != GSL_SUCCESS);
324 
325  /* Iterate root finder */
326  do {
327  gsl_status = gsl_root_fsolver_iterate(s);
328  x = gsl_root_fsolver_root(s);
329  x_lo = gsl_root_fsolver_x_lower(s);
330  x_hi = gsl_root_fsolver_x_upper(s);
331  gsl_status = gsl_root_test_interval (x_lo, x_hi, logs22th13_precision, 0);
332  } while (gsl_status == GSL_CONTINUE && iter < max_iter);
333 
334  /* Save results */
335  x_sens[j] = x;
336  AddToOutput2(t, x);
337 
339 
340  // Setting up output
341 
342  int max_channel_id = 17;
343 
344  char buf[60];
345  char str[60];
346 
347  for(int i=0; i<max_channel_id+1;i++){
348 
349  sprintf(buf,"%dtime2histsim_channel%d.dat",j,i);
350  strcpy(str,buf);
351 
352  FILE* f_out = fopen(str,"w");
353 
354 
355  // double hist_time = 100000;
356  // glbSetRunningTime(EXP_NEAR,0,hist_time);
357 
358  //printf("Running time is; %f",glbGetRunningTime(1,0));
359 
360  glbShowChannelRates(f_out,EXP_NEAR,i,GLB_PRE,GLB_WO_EFF,GLB_WO_BG);
361 
362  fclose(f_out);
363 
364 
365  }
366 
367  }
